def self.search(q, options = {})
  results = []
  
  # add default options
  options = DEFAULT_OPTS.merge(options)
  
  # pull results from twitter
  for page in 1..(options[:rpp]/MAX_RPP).ceil
    results += super(q, options.merge(:page => page, :rpp => MAX_RPP)).results
  end

  results
end
